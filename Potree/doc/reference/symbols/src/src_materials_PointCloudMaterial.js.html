 12 var PointCloudIlluminationMode = {
 13 		FLAT 	: {value:0, name: "Flat"},
 14 		PHONG	: {value:1, name: "Phong"},
 15 		NORMALS	: {value:2, name: "Normals"}
 16 }
 17 
 18 var PointCloudRenderMode = {
 19 		FIXED_CIRCLE 	: {value:0, name: "fixed circle"},
 20 		WEIGHTED_CIRCLE	: {value:1, name: "weighted circle"},
 21 		GAUSSIAN_SPLAT 	: {value:2, name: "gaussian splat"},
 22 		FILL_HOLES 		: {value:3, name: "fill holes"},
 23 }
 24 
 25 /**
 26  * Eierlegende wollmilchsau.
 27  * combines different materials.
 28  * 
 29  * @param name
 30  * @class
 31  * @augments Material
 32  * @author Markus Sch�tz
 33  */
 34 function PointCloudMaterial(name){
 35 	Material.call(this, name);
 36 	
 37 	this.illuminationMode = PointCloudIlluminationMode.FLAT;
 38 	this.renderMode = PointCloudRenderMode.WEIGHTED_CIRCLE;
 39 	
 40 	this.phongMaterial = new PhongMaterial();
 41 	this.gaussMaterial = null;
 42 	this.weightedMaterial = new WeightedPointSizeMaterial();
 43 	this.fixedMaterial = new FixedPointSizeMaterial();
 44 	this.normalsMaterial = new NormalsMaterial();
 45 	this.fillHoleMaterial = null;
 46 	
 47 	this.activeMaterial = this.weightedMaterial;
 48 }
 49 
 50 PointCloudMaterial.prototype = new Material(inheriting);
 51 
 52 PointCloudMaterial.prototype.render = function(mno, mnoSceneNode, camera){
 53 	
 54 	if(Potree.isLowQualityMode && this.activeMaterial != this.phongMaterial){
 55 		this.weightedMaterial.render(mno, mnoSceneNode, camera);
 56 	}else{
 57 		this.activeMaterial.render(mno, mnoSceneNode, camera);
 58 	}
 59 };
 60 
 61 PointCloudMaterial.prototype.getGaussMaterial = function(){
 62 	if(this.gaussMaterial == null){
 63 		this.gaussMaterial = new GaussSplatMaterial();
 64 	}
 65 	
 66 	return this.gaussMaterial;
 67 }
 68 
 69 PointCloudMaterial.prototype.getFillHoleMaterial = function(){
 70 	if(this.fillHoleMaterial == null){
 71 		this.fillHoleMaterial = new FillHoleMaterial();
 72 	}
 73 	
 74 	return this.fillHoleMaterial;
 75 }

103 PointCloudMaterial.prototype.updateActiveMaterial = function(){
104 	if(this.illuminationMode == PointCloudIlluminationMode.FLAT){
105 		if(this.renderMode == PointCloudRenderMode.FIXED_CIRCLE){
106 			this.activeMaterial = this.fixedMaterial;
107 		}else if(this.renderMode == PointCloudRenderMode.WEIGHTED_CIRCLE){
108 			this.activeMaterial = this.weightedMaterial;
109 		}else if(this.renderMode == PointCloudRenderMode.GAUSSIAN_SPLAT){
110 			this.activeMaterial = this.getGaussMaterial();
111 		}
112 	}else if(this.illuminationMode == PointCloudIlluminationMode.PHONG){
113 		if(this.renderMode == PointCloudRenderMode.GAUSSIAN_SPLAT){
114 			this.activeMaterial = this.getGaussMaterial();
115 		}else{
116 			this.activeMaterial = this.phongMaterial;
117 		}
118 	}else if(this.illuminationMode == PointCloudIlluminationMode.NORMALS){
119 		this.activeMaterial = this.normalsMaterial;
120 	}
121 	
122 	if(this.renderMode == PointCloudRenderMode.FILL_HOLES){
123 		this.activeMaterial = this.getFillHoleMaterial();
124 	}
125 }

127 PointCloudMaterial.prototype.setSigmaFactor = function(sigmaFactor){
128 	this.gaussMaterial.sigmaFactor = sigmaFactor;
129 }
130 
131 PointCloudMaterial.prototype.setPointSize = function(pointSize){
132 	if(this.gaussMaterial != null){
133 		this.getGaussMaterial().pointSize = pointSize;
134 	}
135 	this.weightedMaterial.pointSize = pointSize;
136 	this.phongMaterial.pointSize = pointSize;
137 	this.fixedMaterial.pointSize = pointSize;
138 	this.normalsMaterial.pointSize = pointSize;
139 	
140 	if(this.fillHoleMaterial != null){
141 		this.getFillHoleMaterial().pointSize = pointSize;
142 	}
143 }
144 
145 PointCloudMaterial.prototype.setBlendDepth = function(blendDepth){
146 	this.gaussMaterial.blendDepth = blendDepth;
147 }

149 PointCloudMaterial.prototype.setGlossiness = function(glossiness){
150 	this.phongMaterial.glossiness = glossiness;
