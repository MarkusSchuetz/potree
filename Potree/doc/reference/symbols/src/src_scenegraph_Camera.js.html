 12 /**
 13  * @class
 14  * @augments SceneNode
 15  * 
 16  * @Property {float} farClipPlane abc
 17  */
 18 function Camera(name) {
 19 	SceneNode.call(this, name);
 20 	this.nearClipPlane = 0.1;
 21 	this._farClipPlane = 1000.0;
 22 	this._fieldOfView = 60.0;
 23 	this._aspectRatio = 1;
 24 	this._viewMatrix = null;
 25 	this._projectionMatrix = null;
 26 	this.updateViewMatrix();
 27 	this.updateProjectionMatrix();
 28 
 29 }
 30 
 31 Camera.prototype = new SceneNode(inheriting);
 32 Camera.base = SceneNode.prototype;
 33 
 34 Object.defineProperties(Camera.prototype, {
 35 	"farClipPlane" : {
 36 		set: function(farClipPlane){
 37 			this._farClipPlane = farClipPlane;
 38 			this.updateProjectionMatrix();
 39 		},
 40 		get: function(){
 41 			return this._farClipPlane;
 42 		},
 43 	"transform": {
 44 		set: function(transform){
 45 			this._transform = transform;
 46 			this.updateViewMatrix();
 47 		},
 48 		get: function(){
 49 			return this._transform;
 50 		}
 51 	},
 52 	"frustum": {
 53 		get: function(){
 54 			return Frustum.fromCamera(this);
 55 		}
 56 	},
 57 	"projectionMatrix": {
 58 		get: function(){
 59 			return this._projectionMatrix;
 60 		}
 61 	},
 62 	"viewMatrix": {
 63 		get: function(){
 64 			return this._viewMatrix;
 65 		},
 66 		set: function(viewMatrix){
 67 			this._viewMatrix = viewMatrix;
 68 		}
 69 	},
 70 	"fieldOfView": {
 71 		set: function(fieldOfView){
 72 			this._fieldOfView = fieldOfView;
 73 			this.updateProjectionMatrix();
 74 		},
 75 		get: function(){
 76 			return this._fieldOfView;
 77 		}
 78 	},
 79 	"aspectRatio": {
 80 		set: function(aspectRatio){
 81 			this._aspectRatio = aspectRatio;
 82 			this.updateProjectionMatrix();
 83 		},
 84 		get: function(){
 85 			return this._aspectRatio;
 86 		}
 87 	}
 88 });
 89 
 90 Camera.prototype.updateProjectionMatrix = function(){
 91 	this._projectionMatrix = M4x4.makePerspective(this.fieldOfView, this.aspectRatio,
 92 			this.nearClipPlane, this.farClipPlane);
 93 };
 94 
 95 /**
 96  * calculates the current view matrix and stores it in _viewMatrix. use .viewMatrix if you want to get the matrix.
 97  */
 98 Camera.prototype.updateViewMatrix = function(){
 99 	this.viewMatrix = this.getInverseGlobalTransformation();
100 };
101 
102 Camera.prototype.translate = function(x, y, z){
103 	Camera.base.translate.call(this, x, y, z);
104 	this.updateViewMatrix();
105 };
