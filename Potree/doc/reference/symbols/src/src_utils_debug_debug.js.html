 12 /**
 13  * 
 14  * @param parentID
 15  * @class
 16  */
 17 function DebugView(parentID){
 18 	parentComponent = $(parentID);
 19 	
 20 	parentComponent.innerHTML = load_binary_resource("src/utils/debug/debug.html");
 21 	
 22 	this.values = new Object();
 23 	this.tableRows = new Object();
 24 	this.tableColumnsValue = new Object(); 
 25 }
 26 
 27 DebugView.prototype.set = function(key, value){
 28 	if(this.values[key] == null){
 29 		var elTr = document.createElement("tr");
 30 		var elKey = document.createElement("td");
 31 		var elValue = document.createElement("td");
 32 		elKey.innerHTML = key;
 33 		elValue.innerHTML = value;
 34 		
 35 		this.tableRows[key] = elTr;
 36 		this.tableColumnsValue[key] = elValue;
 37 		
 38 		elTr.appendChild(elKey);
 39 		elTr.appendChild(elValue);
 40 		$('debug_table').appendChild(elTr);
 41 	}
 42 	
 43 	this.values[key] = value;
 44 	this.tableColumnsValue[key].innerHTML = value;
 45 };
 46 
 47 DebugView.prototype.get = function(key){
 48 	return this.values[key];
 49 }
