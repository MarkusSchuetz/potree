133 Potree.isWebGLEnabled = function(canvas){
134 	
135 	var names = [ "webgl", "experimental-webgl", "moz-webgl", "webkit-3d" ];
136 	for ( var i = 0; names.length > i; i++) {
137 		try {
138 			gl = canvas.getContext(names[i], {
139 				antialias : false
140 			});
141 			if (gl) {
142 				break;
143 			}
144 		} catch (e) {
145 		}
146 	}
147 	if (!gl) {
148 		Logger.error("No known OpenGL context detected! Is it enabled?");
149 		return false;
150 	}
151 	
152 	return true;
153 };

155 /**
156  * 
157  * @param canvas the canvas element for rendering.
158  * Potree can only be intialized once.
159  * @returns
160  */
161 Potree.init = function(canvas) {
162 	if(Potree.initialized){
163 		Logger.error("Potree has already been initialized");
164 		return true;
165 	}
166 	
167 	Potree.canvas = canvas;
168 	Potree.currentScene = new Scene("default");
169 	
170 	if(!Potree.initGL()){
171 		// init failed -> display error message
172 		var soSorry = document.createElement("div");
173 		var msg = "<br>Could not create a WebGL context. ";
174 		msg += "<br><br>Try using <a href='http://www.mozilla.org'>Firefox</a> ";
175 		msg += "or <a href='www.google.com/chrome/'>Chrome</a>.";
176 		msg += "<br>Other WebGL enabled browsers are not supported but they might work as well.";
177 		soSorry.innerHTML = msg;
178 		soSorry.style.fontSize = "large";
179 		soSorry.style.fontWeight = "bold";
180 		soSorry.style.color = "#FFF";
181 		soSorry.style.textShadow = "black 0 0 4px, black 0 0 4px, black 0 0 4px, black 0 0 4px, black 0 0 4px, black 0 0 4px";
182 		soSorry.style.textAlign = "center";
183 		soSorry.style.verticalAlign = "bottom";
184 		soSorry.style.height = "100%";
185 		canvas.parentNode.replaceChild(soSorry, canvas);
186 		
187 		return false;
188 	}
189 	
190 	{// register mouse and key listener
191 		var mousewheelevt=(/Firefox/i.test(navigator.userAgent))? "DOMMouseScroll" : "mousewheel"; //FF doesn't recognize mousewheel as of FF3.x
192 		if (document.attachEvent){ //if IE (and Opera depending on user setting)
193 		    document.attachEvent("on"+mousewheelevt,MouseListener.mouseWheel);
194 		}else if (document.addEventListener){ //WC3 browsers
195 		    document.addEventListener(mousewheelevt, MouseListener.mouseWheel, false);
196 		}
197 		document.onkeydown = KeyListener.keyDown;
198 		document.onkeyup = KeyListener.keyUp;
199 		document.onkeypress = KeyListener.keyPress;
200 		document.onmousedown = MouseListener.mouseDown;
201 		document.onmouseup = MouseListener.mouseUp;
202 		document.onmousemove = MouseListener.mouseMove;
203 	}
204 	
205 	
206 	{// install cam handler
207 		Potree.camHandler = new FreeFlightCamHandler(Potree.currentScene.activeCamera);
208 		canvas.onfocus = function(){
209 			KeyListener.addListener(Potree.camHandler);
210 			MouseListener.addListener(Potree.camHandler);
211 		};
212 	
213 		canvas.onblur= function(){
214 			KeyListener.removeListener(Potree.camHandler);
215 			MouseListener.removeListener(Potree.camHandler);
216 		};
217 	}
218 	
219 	Potree.initialized = true;
220 	return true;
221 };

223 /**
224  * creates the WebGL context
225  * 
226  */
227 Potree.initGL = function() {
228 	
229 	viewportWidth = Potree.canvas.width;
230 	viewportHeight = Potree.canvas.height;
231 
232 	var names = [ "webgl", "experimental-webgl", "moz-webgl", "webkit-3d" ];
233 	for ( var i = 0; names.length > i; i++) {
234 		try {
235 			gl = Potree.canvas.getContext(names[i], {
236 				antialias : false
237 			});
238 			if (gl) {
239 				break;
240 			}
241 		} catch (e) {
242 		}
243 	}
244 	if (!gl) {
245 		Logger.error("No known OpenGL context detected! Is it enabled?");
246 		return false;
247 	}
248 
249 	if(Potree.useDebugContext){
250 		 logWarn("using WebGLDebugUtils - debugcontext. Performance may suffer.");
251 		 gl = WebGLDebugUtils.makeDebugContext(gl);
252 	}
253 	
254 	// extensions
255 	if (!gl.getExtension("OES_texture_float")) {
256 	    Logger.error("Requires OES_texture_float extension");
257 	}
258 		
259 	// basic settings
260 	// =================================================
261 	var cColor = Potree.Settings.backgroundColor;
262 	gl.clearColor(cColor[0], cColor[1], cColor[2], cColor[3]);
263 	gl.enable(gl.DEPTH_TEST);
264 	gl.enable(gl.BLEND);
265 //	gl.enable(gl.CULL_FACE);
266 //	gl.frontFace(gl.CW);
267 	gl.blendFunc(gl.SRC_ALPHA, gl.ONE_MINUS_SRC_ALPHA);
268 
269 	Potree.testFBO = new Framebuffer(Potree.canvas.width, Potree.canvas.height);
270 	
271 	this.gridSceneNode = Potree.createGridNode();
272 
273 	return true;
274 };

276 /**
277  * a grid at the origin.
278  * 
279  * @returns {SceneNode}
280  */
281 Potree.createGridNode = function createGridNode() {
282 	// materials
283 	var matXAxis = new FlatMaterial("xAxis", [1.0, 0.0, 0.0, 1.0]);
284 	var matYAxis = new FlatMaterial("yAxis", [0.0, 1.0, 0.0, 1.0]);
285 	var matZAxis = new FlatMaterial("zAxis", [0.0, 0.0, 1.0, 1.0]);
286 //	var gridMaterial = new FlatMaterial("grid");
287 //	gridMaterial.setColor(new Colour(0.7, 0.7, 0.7, 1.0));
288 	
289 	var gridNode = new SceneNode("gridNode", null);
290 	var mGrid = MeshUtils.createGrid(1, 20, 20);
291 	var nGrid = new MeshNode("grid", mGrid, gridNode);
292 
293 	// unfortunately, lineWidth is not supported/does not work so create 3 lines
294 	// instead of one thick line
295 	{
296 		var mXAxis = MeshUtils.createLine( [ 0, 0, 0 ], [ 1, 0, 0 ]);
297 		mXAxis.setMaterial(matXAxis);
298 		var nXAxis = new MeshNode("xAxis", mXAxis, gridNode);
299 		var nXAxis1 = new MeshNode("xAxis1", mXAxis, gridNode);
300 		nXAxis1.translate(0, 0, 0.0001);
301 		var nXAxis2 = new MeshNode("xAxis2", mXAxis, gridNode);
302 		nXAxis2.translate(0, 0, -0.0001);
303 	}
304 
305 	{
306 		var mYAxis = MeshUtils.createLine( [ 0, 0, 0 ], [ 0, 1, 0 ]);
307 		mYAxis.setMaterial(matYAxis);
308 		var nYAxis = new MeshNode("yAxis", mYAxis, gridNode);
309 		var nYAxis1 = new MeshNode("yAxis1", mYAxis, gridNode);
310 		nYAxis1.translate(0, 0, 0.0001);
311 		var nYAxis2 = new MeshNode("yAxis2", mYAxis, gridNode);
312 		nYAxis2.translate(0, 0, -0.0001);
313 	}
314 
315 	{
316 		var mZAxis = MeshUtils.createLine( [ 0, 0, 0 ], [ 0, 0, 1 ]);
317 		mZAxis.setMaterial(matZAxis);
318 		var nZAxis = new MeshNode("zAxis", mZAxis, gridNode);
319 		var nZAxis1 = new MeshNode("zAxis1", mZAxis, gridNode);
320 		nZAxis1.translate(0.0001, 0, 0);
321 		var nZAxis2 = new MeshNode("zAxis2", mZAxis, gridNode);
322 		nZAxis2.translate(-0.0001, 0, 0);
323 	}
324 	
325 	return gridNode;
326 };
