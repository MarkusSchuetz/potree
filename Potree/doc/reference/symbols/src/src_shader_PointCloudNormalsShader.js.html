 11 /**
 12  * @class
 13  * @augments Shader
 14  */
 15 function PointCloudNormalsShader(name){
 16 	if (arguments[0] === inheriting) return;
 17 	Shader.call(this, name, "pointsPhong.vs", "normals.fs");
 18 	
 19 }
 20 
 21 PointCloudNormalsShader.prototype = new Shader(inheriting);
 22 
 23 PointCloudNormalsShader.prototype.initUniforms = function(){
 24 	this.uWorld = gl.getUniformLocation(this.program, "world");
 25 	this.uView = gl.getUniformLocation(this.program, "view");
 26 	this.uProjection = gl.getUniformLocation(this.program, "proj");
 27 	this.uOpacity = gl.getUniformLocation(this.program, "uOpacity");
 28 	this.uPointSizeMultiplicator = gl.getUniformLocation(this.program, "uPointSizeMultiplicator");
 29 	this.uViewPos = gl.getUniformLocation(this.program, "uViewPos");
 30 };
 31 
 32 PointCloudNormalsShader.prototype.initAttributes = function(){
 33 	this.aVertexPosition = gl.getAttribLocation(this.program, "aVertexPosition");
 34 	this.aVertexColour = gl.getAttribLocation(this.program, "aVertexColour");
 35 	this.aNormal = gl.getAttribLocation(this.program, "aNormal");
 36 };
