 12 /**
 13  * @class Loads mno files and returns a PointcloudOctree
 14  * for a description of the mno binary file format, read mnoFileFormat.txt
 15  * 
 16  * @author Markus Sch�tz
 17  */
 18 function POCLoader(){
 19 	
 20 }
 21  
 22 /**
 23  * @return a point cloud octree with the root node data loaded. 
 24  * loading of descendants happens asynchronously when they're needed
 25  * 
 26  * @param file the xml mno file
 27  * @param loadingFinishedListener executed after loading the binary has been finished
 28  */
 29 POCLoader.prototype.load = function load(file) {
 30 	this.mno = new PointcloudOctree();
 31 	var xhr = new XMLHttpRequest();
 32 	xhr.open('GET', file, false);
 33 	xhr.send(null);
 34 	if(xhr.status == 200 || xhr.status == 0){
 35 		var fMno = JSON.parse(xhr.responseText);
 36 		
 37 		this.mno.octreeDir = file + "/../data";
 38 		var pointAttributes = POCLoader.loadPointAttributes(fMno);
 39 		this.mno.setPointAttributes(pointAttributes);
 40 		
 41 		{ // load Root
 42 			var mRoot = new PointcloudOctreeNode("r", this.mno);
 43 			var aabb = new AABB();
 44 			aabb.setDimensionByMinMax(
 45 					[ fMno.boundingBox.lx, fMno.boundingBox.ly, fMno.boundingBox.lz ],  
 46 					[ fMno.boundingBox.ux, fMno.boundingBox.uy, fMno.boundingBox.uz ]);
 47 			mRoot.setAABB(aabb);
 48 			mRoot.points = fMno.hierarchy[0][1];
 49 			this.mno.rootNode = mRoot;
 50 		}
 51 		
 52 		// load remaining hierarchy
 53 		for( var i = 1; i < fMno.hierarchy.length; i++){
 54 			var nodeName = fMno.hierarchy[i][0];
 55 			var points = fMno.hierarchy[i][1];
 56 			var mNode = new PointcloudOctreeNode(nodeName, this.mno);
 57 			mNode.points = points;
 58 			this.mno.rootNode.addChild(mNode);
 59 			var childIndex = mNode.name.charAt(mNode.name.length-1);
 60 			var childAABB = POCLoader.createChildAABB(mNode.parent.aabb, childIndex);
 61 			mNode.setAABB(childAABB);
 62 		}
 63 		
 64 	}
 65 	
 66 	return this.mno;
 67 };
 68 
 69 POCLoader.loadPointAttributes = function(mno){
 70 	var fpa = mno.pointAttributes;
 71 	var pa = {
 72 		'numAttributes' : fpa.length,
 73 		'bytesPerPoint' : 0,
 74 		'attributes' : {}
 75 	};
 76 	
 77 	for(var i = 0; i < fpa.length; i++){
 78 		var pointAttribute = PointAttributes[fpa[i]];
 79 		pa.attributes[i] = pointAttribute; 
 80 		var bytes = pointAttribute.type.size * pointAttribute.numElements;
 81 		pa.bytesPerPoint += bytes;
 82 	}
 83 	
 84 	return pa;
 85 };

 88 /**
 89  * creates an aabb that covers the area of the childnode at childIndex
 90  * 
 91  * @param aabb aabb of the parent
 92  * @param childIndex index of the childs region
 93  * @returns {AABB}
 94  */
 95 POCLoader.createChildAABB = function(aabb, childIndex){
 96 	var min = aabb.objectSpaceMin;
 97 	var max = aabb.objectSpaceMax;
 98 	var caabb = new AABB();
 99 	// (aabb.max - aabb.minPos) * 0.5
100 	var dHalfLength = V3.scale(V3.sub(max,min), 0.5);
101 	var xHalfLength = [ dHalfLength[0], 0, 0 ];
102 	var yHalfLength = [ 0, dHalfLength[1], 0 ];
103 	var zHalfLength = [ 0, 0, dHalfLength[2] ];
104 
105 	{
106 		var cmin = min;
107 		// max = min + (aabb.max - aabb.min) * 0.5;
108 		var cmax = V3.add(min, dHalfLength);
109 		var caabb = new AABB();
110 		caabb.setDimensionByMinMax(cmin, cmax);
111 	}
112 
113 	if (childIndex == 1) {
114 		var min = V3.add(caabb.min, zHalfLength);
115 		var max = V3.add(caabb.max, zHalfLength);
116 		caabb.setDimensionByMinMax(min, max);
117 		caabb.setColor([0.0, 0.0, 1.0, 1.0]);
118 	}else if (childIndex == 3) {
119 		var min = V3.add(V3.add(caabb.min, zHalfLength), yHalfLength);
120 		var max = V3.add(V3.add(caabb.max, zHalfLength), yHalfLength);
121 		caabb.setDimensionByMinMax(min, max);
122 		caabb.setColor([1.0, 1.0, 0.0, 1.0]);
123 	}else if (childIndex == 0) {
124 		caabb.setColor([1.0, 0.0, 0.0, 1.0]);
125 	}else if (childIndex == 2) {
126 		var min = V3.add(caabb.min, yHalfLength);
127 		var max = V3.add(caabb.max, yHalfLength);
128 		caabb.setDimensionByMinMax(min, max);
129 		caabb.setColor([0.0, 1.0, 0.0, 1.0]);
130 	}else if (childIndex == 5) {
131 		var min = V3.add(V3.add(caabb.min, zHalfLength), xHalfLength);
132 		var max = V3.add(V3.add(caabb.max, zHalfLength), xHalfLength);
133 		caabb.setDimensionByMinMax(min, max);
134 		caabb.setColor([1.0, 1.0, 1.0, 1.0]);
135 	}else if (childIndex == 7) {
136 		var min = V3.add(caabb.min, dHalfLength);
137 		var max = V3.add(caabb.max, dHalfLength);
138 		caabb.setDimensionByMinMax(min, max);
139 		caabb.setColor([0.0, 0.0, 0.0, 1.0]);
140 	}else if (childIndex == 4) {
141 		var min = V3.add(caabb.min, xHalfLength);
142 		var max = V3.add(caabb.max, xHalfLength);
143 		caabb.setDimensionByMinMax(min, max);
144 		caabb.setColor([1.0, 0.0, 1.0, 1.0]);
145 	}else if (childIndex == 6) {
146 		var min = V3.add(V3.add(caabb.min, xHalfLength), yHalfLength);
147 		var max = V3.add(V3.add(caabb.max, xHalfLength), yHalfLength);
148 		caabb.setDimensionByMinMax(min, max);
149 		caabb.setColor([0.0, 1.0, 1.0, 1.0]);
150 	}
151 	
152 	return caabb;
153 }
