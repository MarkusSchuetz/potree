 12 /**
 13  * @class
 14  * @augments Shader
 15  */
 16 function NormalizationShader(name){
 17 	if (arguments[0] === inheriting) return;
 18 	Shader.call(this, name, "drawTexture.vs", "normalization.fs");
 19 }
 20 
 21 NormalizationShader.prototype = new Shader(inheriting);
 22 
 23 NormalizationShader.prototype.initUniforms = function(){
 24 	this.uGaussSplatTex = gl.getUniformLocation(this.program, "uGaussSplatTex");
 25 	this.uDepthTex = gl.getUniformLocation(this.program, "uDepthTex");
 26 };
 27 
 28 NormalizationShader.prototype.initAttributes = function(){
 29 	this.aVertexPosition = gl.getAttribLocation( this.program, "aVertexPosition");
 30 	this.aTexcoord = gl.getAttribLocation( this.program, "aTexcoord");
 31 };
