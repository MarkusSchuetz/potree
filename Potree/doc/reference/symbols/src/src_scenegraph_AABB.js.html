 12 /**
 13  * 
 14  * @class
 15  * Axis aligned bounding box
 16  * 
 17  * 
 18  * <pre>
 19  *    7----6        y
 20  *   /    /|        |
 21  *  4----5 |        |
 22  *  | 3--|-2        |______ x  
 23  *  |/   |/       .�
 24  *  0----1       �z
 25  * </pre>
 26  *
 27  * @see http://www.lighthouse3d.com/tutorials/view-frustum-culling/geometric-approach-testing-boxes-ii/
 28  */
 29 function AABB(){
 30 	this.material = MaterialManager.getMaterial("aabb");
 31 	
 32 	this.points = 24;
 33 	this.color = [ 1.0, 0.0, 0.0, 1.0];
 34 	
 35 	// transformed minimum and maximum coordinates
 36 	this.min = V3.$(0,0,0);
 37 	this.max = V3.$(0,0,0);
 38 	this.center = null;
 39 	this.radius = null;
 40 	
 41 	this.vbo = null;
 42 	
 43 	this.transform = M4x4.I;
 44 	
 45 	// object space coordinates. of the bounding box. 
 46 	this.p0 = null;
 47 	this.p1 = null;
 48 	this.p2 = null;
 49 	this.p3 = null;
 50 	this.p4 = null;
 51 	this.p5 = null;
 52 	this.p6 = null;
 53 	this.p7 = null;
 54 	this.objectSpaceMin = V3.$(0,0,0);
 55 	this.objectSpaceMax = V3.$(0,0,0);
 56 }

 58 /**
 59  * set object space coordinates of this bounding box
 60  * 
 61  * @param p0
 62  * @param p1
 63  * @param p2
 64  * @param p3
 65  * @param p4
 66  * @param p5
 67  * @param p6
 68  * @param p7

 70 AABB.prototype.setDimension = function(p0, p1, p2, p3, p4, p5, p6, p7) {
 71 	this.p0 = p0;
 72 	this.p1 = p1;
 73 	this.p2 = p2;
 74 	this.p3 = p3;
 75 	this.p4 = p4;
 76 	this.p5 = p5;
 77 	this.p6 = p6;
 78 	this.p7 = p7;
 79 	this.update();
 80 };

 82 /**
 83  * set object space coordinates by providing the minimum and the maximum vertex of the aabb.
 84  * 
 85  * @param min
 86  * @param max
 87  */
 88 AABB.prototype.setDimensionByMinMax = function(min, max){
 89 	this.p0 = [min[0], min[1], max[2]];
 90 	this.p1 = [max[0], min[1], max[2]];
 91 	this.p2 = [max[0], min[1], min[2]];
 92 	this.p3 = [min[0], min[1], min[2]];
 93 	this.p4 = [min[0], max[1], max[2]];
 94 	this.p5 = [max[0], max[1], max[2]];
 95 	this.p6 = [max[0], max[1], min[2]];
 96 	this.p7 = [min[0], max[1], min[2]];
 97 	this.update();
 98 };

112 AABB.tmpVec = V3.$(0,0,0);
113 /**
114  * calculates min, max, center and radius of the transformed AABB. p0 to p7 will be kept in object space.
115  */
116 AABB.prototype.update = function update(){
117 	
118 	// calculate the minimum of the box
119 	this.objectSpaceMin[0] = Math.min(this.p0[0], this.p1[0], this.p2[0], this.p3[0], this.p4[0], this.p5[0], this.p6[0], this.p7[0]);
120 	this.objectSpaceMin[1] = Math.min(this.p0[1], this.p1[1], this.p2[1], this.p3[1], this.p4[1], this.p5[1], this.p6[1], this.p7[1]);
121 	this.objectSpaceMin[2] = Math.min(this.p0[2], this.p1[2], this.p2[2], this.p3[2], this.p4[2], this.p5[2], this.p6[2], this.p7[2]);
122 	
123 	// calculate maximum of the box
124 	this.objectSpaceMax[0] = Math.max(this.p0[0], this.p1[0], this.p2[0], this.p3[0], this.p4[0], this.p5[0], this.p6[0], this.p7[0]);
125 	this.objectSpaceMax[1] = Math.max(this.p0[1], this.p1[1], this.p2[1], this.p3[1], this.p4[1], this.p5[1], this.p6[1], this.p7[1]);
126 	this.objectSpaceMax[2] = Math.max(this.p0[2], this.p1[2], this.p2[2], this.p3[2], this.p4[2], this.p5[2], this.p6[2], this.p7[2]);
127 	
128 	// calculate transformed box coordinates
129 	var p0 = V3.transform(this.p0, this.transform);
130 	var p1 = V3.transform(this.p1, this.transform);
131 	var p2 = V3.transform(this.p2, this.transform);
132 	var p3 = V3.transform(this.p3, this.transform);
133 	var p4 = V3.transform(this.p4, this.transform);
134 	var p5 = V3.transform(this.p5, this.transform);
135 	var p6 = V3.transform(this.p6, this.transform);
136 	var p7 = V3.transform(this.p7, this.transform);
137 	
138 	// calculate the minimum of the transformed box
139 	this.min[0] = Math.min(p0[0], p1[0], p2[0], p3[0], p4[0], p5[0], p6[0], p7[0]);
140 	this.min[1] = Math.min(p0[1], p1[1], p2[1], p3[1], p4[1], p5[1], p6[1], p7[1]);
141 	this.min[2] = Math.min(p0[2], p1[2], p2[2], p3[2], p4[2], p5[2], p6[2], p7[2]);
142 	
143 	// calculate maximum of the transformed box
144 	this.max[0] = Math.max(p0[0], p1[0], p2[0], p3[0], p4[0], p5[0], p6[0], p7[0]);
145 	this.max[1] = Math.max(p0[1], p1[1], p2[1], p3[1], p4[1], p5[1], p6[1], p7[1]);
146 	this.max[2] = Math.max(p0[2], p1[2], p2[2], p3[2], p4[2], p5[2], p6[2], p7[2]);
147 	
148 	// center
149 	this.center = V3.add(this.min, this.max);
150 	this.center = V3.scale(this.center, 0.5);
151 	
152 	// radius
153 	V3.sub(this.max, this.min, AABB.tmpVec);
154 	this.radius = V3.length(AABB.tmpVec) / 2.0;
155 }

157 AABB.prototype.updateVBO = function(){
158 	if(this.vbo == null){
159 		this.vbo =  gl.createBuffer();
160 	}
161 	
162 	var sx = this.min[0];
163 	var sy = this.min[1];
164 	var sz = this.min[2];
165 	var gx = this.max[0];
166 	var gy = this.max[1];
167 	var gz = this.max[2];
168 
169 	
170 	var vertices = [
171         sx, sy, sz, gx, sy, sz,
172         sx, sy, sz, sx, gy, sz,
173         sx, sy, sz, sx, sy, gz,
174         gx, gy, gz, sx, gy, gz,
175         gx, sy, gz, gx, gy, gz,
176         gx, gy, gz, gx, gy, sz,
177         
178         sx, gy, sz, sx, gy, gz,
179         sx, gy, gz, sx, sy, gz,
180         sx, sy, gz, gx, sy, gz,
181         sx, gy, sz, gx, gy, sz,
182         gx, sy, sz, gx, gy, sz,
183         gx, sy, sz, gx, sy, gz
184 	];
185 	gl.bindBuffer(gl.ARRAY_BUFFER, this.vbo);
186 	gl.bufferData(gl.ARRAY_BUFFER, new WebGLFloatArray(vertices), gl.STATIC_DRAW);
187 };

189 /**
190  * returns the distance from minPos to maxPos in projected space
191  */
192 AABB.prototype.getOnScreenSize = function(sceneNode, camera){
193 	var world = sceneNode.getGlobalTransformation();
194 	var view = camera.viewMatrix;
195 	var proj = camera.projectionMatrix;
196 	
197 	var worldViewProj = M4x4.mul(proj, M4x4.mul(view, world));
198 	var min = V3.transform(this.min, worldViewProj);
199 	var max = V3.transform(this.max, worldViewProj);
200 	var diff = V3.sub(min, max);
201 	var length = V3.length(diff);
202 	
203 	return length;
204 	
205 };

222 /**
223  * returns the positive coordinate of the transformed bounding box. 
224  * 
225  * @see http://www.lighthouse3d.com/tutorials/view-frustum-culling/geometric-approach-testing-boxes-ii/
226  * @param normal
227  */
228 AABB.prototype.getPositive = function getPositive(normal){
229 	var min = this.min;
230 	var max = this.max;
231 	var p = [min[0], min[1], min[2]];
232 	if (normal[0] >= 0){
233 		p[0] = max[0];
234 	}
235 	if (normal[1] >=0){
236 		p[1] = max[1];
237 	}
238 	if (normal[2] >= 0){
239 		p[2] = max[2];
240 	}
241 	
242 	return p;
243 }
