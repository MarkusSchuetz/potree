 12 /**
 13  * You can call the constructor with either, width and height, or "system".
 14  * When invoking the constructor with "system", the framebuffer will refer to the system/default framebuffer.
 15  * 
 16  * @param {int} width
 17  * @param {int} height
 18  * @class create and handle FramebufferObjects
 19  * @see <a href="http://learningwebgl.com/blog/?p=1786">WebGL Lesson 16</a>
 20  */
 21 function Framebuffer(width, height){
 22 	if (arguments[0] === inheriting) return;
 23 	if(arguments[0] == "system"){
 24 		this.framebuffer = null;
 25 		this.initOtherStuff();
 26 	}else{
 27 		this.initBufferStuff(width, height);
 28 		this.initOtherStuff();
 29 	}
 30 }

 32 /**
 33  * 
 34  * @returns the system framebuffer
 35  */
 36 Framebuffer.getSystemBuffer = function(){
 37 	if(Framebuffer.systemBuffer == null){
 38 		Framebuffer.systemBuffer = new Framebuffer("system");
 39 	}
 40 	
 41 	return Framebuffer.systemBuffer;
 42 }

 44 Framebuffer.getActiveBuffer = function(){
 45 	if(Framebuffer.activeBuffer == null){
 46 		Framebuffer.activeBuffer = Framebuffer.getSystemBuffer();
 47 	}
 48 	
 49 	return Framebuffer.activeBuffer;
 50 }

 66 /**
 67  * Initialize stuff like a vbo for screen quads.
 68  */
 69 Framebuffer.prototype.initOtherStuff = function(){
 70 	
 71 	
 72 	this.vbo = gl.createBuffer();
 73 	this.texcoordvbo = gl.createBuffer();
 74 	gl.bindBuffer(gl.ARRAY_BUFFER, this.vbo);
 75 	var vertices = new Float32Array([
 76 	                                  0,0,0,
 77 	                                  1,0,0,
 78 	                                  1,1,0,
 79 	                                  0,0,0,
 80 	                                  1,1,0,
 81 	                                  0,1,0
 82 	                                  ]);
 83 	gl.bufferData(gl.ARRAY_BUFFER, vertices, gl.STATIC_DRAW);
 84 	
 85 	gl.bindBuffer(gl.ARRAY_BUFFER, this.texcoordvbo);
 86 	var vertices = new Float32Array([
 87 	                                  0,0,
 88 	                                  1,0,
 89 	                                  1,1,
 90 	                                  0,0,
 91 	                                  1,1,
 92 	                                  0,1
 93 	                                  ]);
 94 	gl.bufferData(gl.ARRAY_BUFFER, vertices, gl.STATIC_DRAW);
 95 	
 96 };

 98 /**
 99  * change size of screenQuad.
100  * 
101  * @param {[x,y]} start x and y must be values between 0 and 1
102  * @param {[x,y]} end x and y must be values between 0 and 1
103  */
104 Framebuffer.prototype.updateScreenQuad = function(start, end){
105 	gl.bindBuffer(gl.ARRAY_BUFFER, this.vbo);
106 	var vertices = new Float32Array([
107 	                                  start[0],start[1],0,
108 	                                  end[0],start[1],0,
109 	                                  end[0],end[1],0,
110 	                                  start[0],start[1],0,
111 	                                  end[0],end[1],0,
112 	                                  start[0],end[1],0
113 	                                  ]);
114 	gl.bufferData(gl.ARRAY_BUFFER, vertices, gl.STATIC_DRAW);
115 };

117 /**
118  * removes existing webgl color/render/frame-buffers and creates new ones
119  * 
120  * @param {int} width
121  * @param {int} height
122  */
123 Framebuffer.prototype.initBufferStuff = function(width, height){
124 	if(this.width == width && this.height == height){
125 		return;
126 	}
127 	
128 	this.width = width;
129 	this.height = height;
130 	
131 	// remove exiting buffers
132 	gl.bindFramebuffer(gl.FRAMEBUFFER, null);
133 	if(this.texture != null){
134 		gl.deleteTexture(this.texture.glid);
135 		this.texture = null;
136 	}
137 	if(this.renderbuffer != null){
138 		gl.deleteRenderbuffer(this.renderbuffer);
139 		this.renderbuffer = null;
140 	}
141 	if(this.framebuffer != null){
142 		gl.deleteFramebuffer(this.framebuffer);
143 		this.framebuffer = null;
144 	}
145 	
146 	// create new buffers
147 	this.framebuffer = gl.createFramebuffer();
148 	this.texture = new Texture();
149 	this.texture.glid = gl.createTexture();
150 	this.renderbuffer = gl.createRenderbuffer();
151 	
152 	// framebuffer
153 	gl.bindFramebuffer(gl.FRAMEBUFFER, this.framebuffer);
154 	this.framebuffer.width = width;
155 	this.framebuffer.height = height;
156 	
157 	// colorbuffer
158 	this.initColorbuffer();
159     
160     // depthbuffer
161 	gl.bindRenderbuffer(gl.RENDERBUFFER, this.renderbuffer);
162 	gl.renderbufferStorage(gl.RENDERBUFFER, gl.DEPTH_COMPONENT16, this.framebuffer.width, this.framebuffer.height);
163     
164 	// assemble buffers
165 	gl.framebufferTexture2D(gl.FRAMEBUFFER, gl.COLOR_ATTACHMENT0, gl.TEXTURE_2D, this.texture.glid, 0);
166     gl.framebufferRenderbuffer(gl.FRAMEBUFFER, gl.DEPTH_ATTACHMENT, gl.RENDERBUFFER, this.renderbuffer);
167 
168     this.checkBuffer();
169     
170     gl.bindTexture(gl.TEXTURE_2D, null);
171     gl.bindRenderbuffer(gl.RENDERBUFFER, null);
172     gl.bindFramebuffer(gl.FRAMEBUFFER, null);
173 };

175 /**
176  * initializes the colourbuffer that'll be used as COLOR_ATTACHMENT0
177  */
178 Framebuffer.prototype.initColorbuffer = function(){
179 	gl.bindTexture(gl.TEXTURE_2D, this.texture.glid);
180 	gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, gl.NEAREST);
181 	gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.NEAREST);
182 	gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_S, gl.CLAMP_TO_EDGE);
183 	gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_T, gl.CLAMP_TO_EDGE);
184     gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, this.framebuffer.width, this.framebuffer.height, 0, gl.RGBA, gl.UNSIGNED_BYTE, null);
185 };

187 /**
188  * check if framebuffer was successfully created. Throws an exception if the buffer is invalid. 
189  */
190 Framebuffer.prototype.checkBuffer = function checkBuffer(){
191 	gl.bindFramebuffer(gl.FRAMEBUFFER, this.framebuffer);
192 	
193 	var status = gl.checkFramebufferStatus(gl.FRAMEBUFFER);
194     switch (status) {
195         case gl.FRAMEBUFFER_COMPLETE:
196             break;
197         case gl.FRAMEBUFFER_INCOMPLETE_ATTACHMENT:
198         	Logger.error("Incomplete framebuffer: FRAMEBUFFER_INCOMPLETE_ATTACHMENT");
199         	throw "";
200             break;
201         case gl.FRAMEBUFFER_INCOMPLETE_MISSING_ATTACHMENT:
202         	Logger.error("Incomplete framebuffer: FRAMEBUFFER_INCOMPLETE_MISSING_ATTACHMENT");
203         	throw "";
204             break;
205         case gl.FRAMEBUFFER_INCOMPLETE_DIMENSIONS:
206         	Logger.error("Incomplete framebuffer: FRAMEBUFFER_INCOMPLETE_DIMENSIONS");
207         	throw "";
208             break;
209         case gl.FRAMEBUFFER_UNSUPPORTED:
210         	Logger.error("Incomplete framebuffer: FRAMEBUFFER_UNSUPPORTED");
211         	throw "";
212             break;
213         default:
214         	Logger.error("Incomplete framebuffer: " + status);
215         	throw "";
216     }
217 };

220 Framebuffer.bindDefault = function(){
221 	Framebuffer.activeBuffer = Framebuffer.getSystemBuffer();
222 	gl.bindFramebuffer(gl.FRAMEBUFFER, null);
223 };

225 /**
226  * binds this framebuffer which makes it the target for all drawCalls and framebuffer related calls.
227  */
228 Framebuffer.prototype.bind = function(){
229 	Framebuffer.activeBuffer = this;
230 	gl.bindFramebuffer(gl.FRAMEBUFFER, this.framebuffer);
231 };

233 /**
234  * 
235  * @param {Texture} texture
236  * @param {[x,y]} start
237  * @param {[x,y]} end
238  * 
239  * @example drawTexture(abc, [0,0], [1,1])
240  */
241 Framebuffer.prototype.drawTexture = function(texture, start, end){
242 	this.bind();
243 	
244 	var mat = ShaderManager.getShader("drawTexture");
245 	gl.useProgram(mat.program);
246 	
247 	this.updateScreenQuad(start, end);
248 
249 	// uniforms
250 	if(this.framebuffer == null){
251 		var canvas = document.getElementById("canvas");
252 		gl.uniform1f(mat.uWidth, canvas.width);
253 		gl.uniform1f(mat.uHeight, canvas.height);
254 	}else{
255 		gl.uniform1f(mat.uWidth, this.width);
256 		gl.uniform1f(mat.uHeight, this.height);
257 	}
258 	// texture
259 	gl.activeTexture(gl.TEXTURE0);
260 	gl.bindTexture(gl.TEXTURE_2D, texture.glid);
261 	gl.uniform1i(mat.uTexture, 0);
262 	// depth
263 	gl.activeTexture(gl.TEXTURE1);
264 	gl.bindTexture(gl.TEXTURE_2D, this.renderbuffer);
265 	gl.uniform1i(mat.uDepth, 0);
266 	
267 	// vertex attributes
268 	gl.enableVertexAttribArray(mat.aVertexPosition);
269 	gl.enableVertexAttribArray(mat.aTexcoords);
270 	
271 	gl.bindBuffer(gl.ARRAY_BUFFER, this.vbo);
272 	gl.vertexAttribPointer(mat.aVertexPosition, 3, gl.FLOAT, false, 0, 0);
273 	
274 	gl.bindBuffer(gl.ARRAY_BUFFER, this.texcoordvbo);
275 	gl.vertexAttribPointer(mat.aTexcoords, 2, gl.FLOAT, false, 0, 0);
276 	
277 	gl.drawArrays(gl.TRIANGLES, 0, 6);
278 	
279 };

281 /**
282  * draws a quad over the whole framebuffer using the provided material.
283  * 
284  * @param {Material} mat
285  */
286 Framebuffer.prototype.drawFullscreenQuad = function(mat){
287 	
288 	gl.useProgram(mat.program);
289 	
290 	
291 	
292 	this.updateScreenQuad([-1,-1], [1,1]);
293 	
294 	
295 
296 	// vertex attributes
297 	gl.enableVertexAttribArray(mat.aVertexPosition);
298 	gl.enableVertexAttribArray(mat.aTexcoords);
299 	
300 	gl.bindBuffer(gl.ARRAY_BUFFER, this.vbo);
301 	gl.vertexAttribPointer(mat.aVertexPosition, 3, gl.FLOAT, false, 0, 0);
302 	
303 	gl.bindBuffer(gl.ARRAY_BUFFER, this.texcoordvbo);
304 	gl.vertexAttribPointer(mat.aTexcoords, 2, gl.FLOAT, false, 0, 0);
305 	
306 	gl.drawArrays(gl.TRIANGLES, 0, 6);
307 	
308 	gl.disableVertexAttribArray(mat.aVertexPosition);
309 	gl.disableVertexAttribArray(mat.aTexcoords);
310 };
