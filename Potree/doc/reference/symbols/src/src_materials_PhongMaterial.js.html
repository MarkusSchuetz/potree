 12 /**
 13  * @class
 14  * @augments Material
 15  * @author Markus Sch�tz
 16  */
 17 function PhongMaterial(name){
 18 	Material.call(this, name);
 19 	this.subMeshPhongShader = new PhongShader();
 20 	this.pointCloudPhongShader = new PointCloudPhongShader();
 21 	
 22 	this.pointSize = 1.0;
 23 	this.glossiness = 50;
 24 }
 25 
 26 PhongMaterial.prototype = new Material(inheriting);
 27 
 28 /**
 29  * 
 30  * 
 31  * @param object may be either a SubMesh or a PointcloudOctree
 32  * @param sceneNode
 33  * @param camera
 34  */
 35 PhongMaterial.prototype.render = function(object, sceneNode, camera){
 36 	if(object instanceof SubMesh){
 37 		this.renderSubMesh(object, sceneNode, camera);
 38 	}else if(object instanceof PointcloudOctree){
 39 		this.renderPointcloudOctree(object, sceneNode, camera);
 40 	}
 41 };
 42 
 43 PhongMaterial.prototype.renderPointcloudOctree = function(mno, mnoSceneNode, camera){
 44 	var shader = this.pointCloudPhongShader;
 45 	var mnoNodes = mno.renderQueue.nodeList;
 46 	
 47 	for(var i = 0; i < mnoNodes.size(); i++){
 48 		var node = mnoNodes[i];
 49 		var pointCloud = node.pointCloud;
 50 		var pointAttributes = mnoSceneNode.mno.pointAttributes;
 51 		var viewPos = camera.getGlobalPosition();
 52 		
 53 //		var numChildren = 0;
 54 //		var visibilityMultiplicator = 0;
 55 //		for(var index in node.children){
 56 //			visibilityMultiplicator += (1-node.children[index].opacity);
 57 //			numChildren++;
 58 //		}
 59 //		visibilityMultiplicator = (2*visibilityMultiplicator / numChildren) + 1;
 60 
 61 		gl.useProgram(shader.program);
 62 		
 63 		{ // uniforms
 64 			gl.uniformMatrix4fv(shader.uWorld, false, mnoSceneNode.getGlobalTransformation());
 65 			gl.uniformMatrix4fv(shader.uView, false, camera.viewMatrix);
 66 			gl.uniformMatrix4fv(shader.uProjection, false, camera.projectionMatrix);
 67 			gl.uniform1f(shader.uPointSizeMultiplicator, node.opacity * this.pointSize );
 68 			gl.uniform3f(shader.uViewPos, viewPos[0], viewPos[1], viewPos[2]);
 69 			gl.uniform1f(shader.glossiness, this.glossiness);
 70 		}
 71 		
 72 		gl.bindBuffer(gl.ARRAY_BUFFER, pointCloud.vbo);
 73 		var offset = 0;
 74 		for(var j = 0; j < pointAttributes.numAttributes; j++){
 75 			var attribute = pointAttributes.attributes[j];
 76 			
 77 			if(attribute.name == PointAttributeNames.POSITION_CARTESIAN){
 78 				gl.enableVertexAttribArray(shader.aVertexPosition);
 79 				gl.vertexAttribPointer(shader.aVertexPosition, 3, gl.FLOAT, false,pointAttributes.bytesPerPoint, offset);
 80 			}else if(attribute.name == PointAttributeNames.COLOR_PACKED){
 81 				if(shader.aVertexColour != null){
 82 					gl.enableVertexAttribArray(shader.aVertexColour);
 83 					gl.vertexAttribPointer(shader.aVertexColour, 3, gl.UNSIGNED_BYTE, false,pointAttributes.bytesPerPoint, offset);
 84 				}
 85 			}else if(attribute.name == PointAttributeNames.NORMAL_FLOATS){
 86 				if(shader.aNormal != null){
 87 					gl.enableVertexAttribArray(shader.aNormal);
 88 					gl.vertexAttribPointer(shader.aNormal, 3, gl.FLOAT, false,pointAttributes.bytesPerPoint, offset);
 89 				}
 90 			}
 91 			offset += attribute.type.size * attribute.numElements;
 92 		}
 93 		
 94 		gl.drawArrays(gl.POINTS, 0, node.points / 5);
 95 		Potree.drawnPoints += node.points;
 96 		Potree.drawCalls += 1;
 97 	}
 98 };

100 PhongMaterial.prototype.renderSubMesh = function(subMesh, meshNode, camera){
101 	var shader = this.subMeshPhongShader;
102 	
103 	var scene = camera.scene;
104 	var mesh = meshNode.mesh;
105 	gl.useProgram(shader.program);
106 
107 	// uniforms
108 	gl.uniformMatrix4fv(shader.uWorld, false, meshNode.getGlobalTransformation());
109 	gl.uniformMatrix4fv(shader.uView, false, camera.viewMatrix);
110 	gl.uniformMatrix4fv(shader.uProjection, false, camera.projectionMatrix);
111 	var viewPos = camera.getGlobalPosition();
112 	gl.uniform3f(shader.uViewPos, viewPos[0], viewPos[1], viewPos[2]);
113 	
114 	// TODO find better solution for lights
115 	if(shader.uLightPos != null){
116 		var light = scene.lights[Object.keys(scene.lights)[0]];
117 		var lightPos = light.getGlobalPosition();
118 		gl.uniform3f(shader.uLightPos, lightPos[0], lightPos[1], lightPos[2]);
119 	}
120 	
121 	// vertex attributes
122 	gl.enableVertexAttribArray(shader.aVertexPosition);
123 	gl.bindBuffer(gl.ARRAY_BUFFER, subMesh.vbos["POSITION"]);
124 	gl.vertexAttribPointer(shader.aVertexPosition, 3, gl.FLOAT, false, 0, 0);
125 	
126 	
127 	if(subMesh.vbos["TEXCOORD_0"] != null && shader.aTextureCoord != null ){
128 		gl.enableVertexAttribArray(shader.aTextureCoord);
129 		gl.bindBuffer(gl.ARRAY_BUFFER, subMesh.vbos["TEXCOORD_0"]);
130 		gl.vertexAttribPointer(shader.aTextureCoord, 2, gl.FLOAT, false, 0, 0);
131 	}
132 	
133 	if(subMesh.vbos["NORMAL"] != null && shader.aNormal != null){
134 		gl.enableVertexAttribArray(shader.aNormal);
135 		gl.bindBuffer(gl.ARRAY_BUFFER, subMesh.vbos["NORMAL"]);
136 		gl.vertexAttribPointer(shader.aNormal, 3, gl.FLOAT, false, 0, 0);
137 	}
138 	
139 	if(subMesh.ibo != null){
140 		gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, subMesh.ibo);
141 		gl.drawElements(mesh.glType, subMesh.indices.length, gl.UNSIGNED_BYTE, 0);
142 		Potree.drawCalls += 1;
143 	}else if(subMesh.vertexCount != null){
144 		gl.lineWidth(10.0);
145 		gl.drawArrays(mesh.glType, 0, subMesh.vertexCount);
146 		Potree.drawCalls += 1;
147 	}
148 };
