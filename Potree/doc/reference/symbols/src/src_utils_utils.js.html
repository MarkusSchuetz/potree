 20 /**
 21  * load a binary resource from the given url.
 22  * 
 23  * @see https://developer.mozilla.org/En/Using_XMLHttpRequest 
 24  * 
 25  */
 26 function load_binary_resource(url) {
 27 	Logger.trace("loading "+url);
 28 	var req = new XMLHttpRequest();
 29 	req.open('GET', url, false);
 30 	// The following line says we want to receive data as Binary and not as Unicode
 31 	req.overrideMimeType('text/plain; charset=x-user-defined');
 32 	req.send(null);
 33 	// when accessing local files, req.status will be 0
 34 	if (req.status != 200 && req.status != 0) {
 35 		Logger.error("req.status: '" + req.status + "'");
 36 		Logger.error("req.readyState: '" + req.readyState + "'");
 37 		return '';
 38 	}
 39 	return req.responseText;
 40 }

 42 /**
 43  * load a binary resource directly into an an ArrayBuffer
 44  * 
 45  * @param url
 46  * @returns an arraybuffer with data from url
 47  */
 48 function loadBinaryResourceIntoArrayBuffer(url) {
 49 	var xhr = new XMLHttpRequest();
 50 	xhr.open('GET', url, false);
 51 	xhr.responseType = 'arraybuffer';
 52 	xhr.overrideMimeType('text/plain; charset=x-user-defined');
 53 	xhr.send(null);
 54 	if (xhr.readyState == 4) {
 55 		// when accessing local files, req.status will be 0
 56 		if (xhr.status == 200 || xhr.status == 0) {
 57 			var buffer = xhr.response;
 58 			return buffer;
 59 		} else {
 60 			alert('Failed to load file! HTTP status: ' + xhr.status);
 61 		}
 62 	}
 63 
 64 	return null;
 65 }

 80 /**
 81  * add separators to large numbers
 82  * 
 83  * @param nStr
 84  * @returns
 85  */
 86 function addCommas(nStr)
 87 {
 88 	nStr += '';
 89 	x = nStr.split('.');
 90 	x1 = x[0];
 91 	x2 = x.length > 1 ? '.' + x[1] : '';
 92 	var rgx = /(\d+)(\d{3})/;
 93 	while (rgx.test(x1)) {
 94 		x1 = x1.replace(rgx, '$1' + ',' + '$2');
 95 	}
 96 	return x1 + x2;
 97 }
