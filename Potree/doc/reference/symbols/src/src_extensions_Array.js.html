 19 /**
 20  * remove all occurences of element in the array
 21  */
 22 Array.prototype.remove=function(element){
 23 	var index = null;
 24 	while((index = this.indexOf(element)) != -1){
 25 		this.splice(index, 1);
 26 	}
 27 };
 28 
 29 Array.prototype.contains = function(element){
 30 	var index = this.indexOf(element);
 31 	return index != -1;
 32 };
